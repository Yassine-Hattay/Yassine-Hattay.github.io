130// Garbage collecting examines all pages in a block which and sums up
131// to a block score. Deleted pages normally gives positive score and
132// used pages normally gives a negative score (as these must be moved).
133// To have a fair wear-leveling, the erase age is also included in score,
134// whose factor normally is the most positive.
135// The larger the score, the more likely it is that the block will
136// picked for garbage collection.
137
138// Garbage collecting heuristics - weight used for deleted pages.
139#define SPIFFS_GC_HEUR_W_DELET (5)
140// Garbage collecting heuristics - weight used for used pages.
141#define SPIFFS_GC_HEUR_W_USED (-1)
142// Garbage collecting heuristics - weight used for time between
143// last erased and erase of this block.
144#define SPIFFS_GC_HEUR_W_ERASE_AGE (50)

236// Enable this to add a temporal file cache using the fd buffer.
237// The effects of the cache is that SPIFFS_open will find the file faster in
238// certain cases. It will make it a lot easier for spiffs to find files
239// opened frequently, reducing number of readings from the spi flash for
240// finding those files.
241// This will grow each fd by 6 bytes. If your files are opened in patterns
242// with a degree of temporal locality, the system is optimized.
243// Examples can be letting spiffs serve web content, where one file is the css.
244// The css is accessed for each html file that is opened, meaning it is
245// accessed almost every second time a file is opened. Another example could be
246// a log file that is often opened, written, and closed.
247// The size of the cache is number of given file descriptors, as it piggybacks
248// on the fd update mechanism. The cache lives in the closed file descriptors.
249// When closed, the fd know the whereabouts of the file. Instead of forgetting
250// this, the temporal cache will keep handling updates to that file even if the
251// fd is closed. If the file is opened again, the location of the file is found
252// directly. If all available descriptors become opened, all cache memory is
253// lost.
254#define SPIFFS_TEMPORAL_FD_CACHE 1
255
256// Temporal file cache hit score. Each time a file is opened, all cached files
257// will lose one point. If the opened file is found in cache, that entry will
258// gain SPIFFS_TEMPORAL_CACHE_HIT_SCORE points. One can experiment with this
259// value for the specific access patterns of the application. However, it must
260// be between 1 (no gain for hitting a cached entry often) and 255.
261#define SPIFFS_TEMPORAL_CACHE_HIT_SCORE 4

263// Enable to be able to map object indices to memory.
264// This allows for faster and more deterministic reading if cases of reading
265// large files and when changing file offset by seeking around a lot.
266// When mapping a file's index, the file system will be scanned for index pages
267// and the info will be put in memory provided by user. When reading, the
268// memory map can be looked up instead of searching for index pages on the
269// medium. This way, user can trade memory against performance.
270// Whole, parts of, or future parts not being written yet can be mapped. The
271// memory array will be owned by spiffs and updated accordingly during garbage
272// collecting or when modifying the indices. The latter is invoked by when the
273// file is modified in some way. The index buffer is tied to the file
274// descriptor.
275#define SPIFFS_IX_MAP 1

299// Types depending on configuration such as the amount of flash bytes

310// Object id type - most significant bit is reserved for index flag. Make sure the
311// size of this type can hold the highest object id on a full system,
312// i.e. 2 + (spiffs_file_system_size / (2*log_page_size))*2
313typedef u16_t spiffs_obj_id;
314// Object span index type. Make sure the size of this type can
315// hold the largest possible span index on the system -
316// i.e. (spiffs_file_system_size / log_page_size) - 1
317typedef u16_t spiffs_span_ix;
